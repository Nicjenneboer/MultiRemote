307 void uart_init(){
308  static bool init_done = false;
309  if( init_done ){
310  return;
311  }
312  init_done = true;
313 
314  // set baudrate
315  uint64_t UBRR_VALUE = ((( 16'000'000 / ( BMPTK_BAUDRATE * 16UL ))) - 1 );
316  UBRR0H = (uint8_t) ( UBRR_VALUE >> 8 );
317  UBRR0L = (uint8_t) UBRR_VALUE;
318 
319  // format : 8 data bits, no parity, 1 stop bit
320  UCSR0C = 0x06;
321 
322  // enable rx and tx
323  UCSR0B = (1<<RXEN0)|(1<<TXEN0);
324 }

353 void HWLIB_WEAK wait_us_asm( int n ){
354  // first int parameter is passd in r24/r25
355  __asm volatile( // clocks
356  "1: cp r1, r24 \t\n" // 1
357  " cpc r1, r25 \t\n" // 1
358  " brge 3f \t\n" // 1
359  " rcall 3f \t\n" // 7
360  " rjmp 2f \t\n" // 2
361  "2: sbiw r24, 0x01 \t\n" // 2
362  " rjmp 1b \t\n" // 2
363  "3: \t\n" // 16 total
364  : : "r" ( n ) // uses (reads) n
365  );
366 
367 }

387 //static inline uint16_t last_low = 0;
388 //static inline uint_fast64_t high = 0;
389 static uint16_t last_low = 0;
390 static uint_fast64_t high = 0;
391 
392 uint_fast64_t now_ticks(){
393 
394  static bool init_done = false;
395  if( ! init_done ){
396  // set up timer 1 without prescaler (input=CPU clock)
397  TCCR1B = 0x01;
398  init_done = true;
399  }
400 
401  uint16_t low = TCNT1L;
402  low |= ( TCNT1H << 8 );
403 
404  if( low < last_low ){
405 
406  // the timer rolled over, so increment the high part
407  high += 0x1ULL << 16;
408  }
409  last_low = low;
410 
411  return ( low | high );
412 }
413 
414 uint64_t now_us(){
415  return now_ticks() / ticks_per_us();
416 }
