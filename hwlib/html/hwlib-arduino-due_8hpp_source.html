520 class d2_36kHz : public hwlib::pin_out {
521  public:
522 
524  d2_36kHz(){
525 
526  // this sets the main clock to 84 MHz
527  hwlib::wait_ms( 1 );
528 
529  // enable the clock to port B
530  PMC->PMC_PCER0 = 1 << ID_PIOB;
531 
532  // disable PIO Control on PB25 and set up for Peripheral B TIOA0
533  PIOB->PIO_PDR = PIO_PB25;
534  PIOB->PIO_ABSR |= PIO_PB25;
535 
536  // enable output on B25
537  PIOB->PIO_OER = PIO_PB25;
538 
539  // enable the clock to the TC0 (peripheral # 27)
540  PMC->PMC_PCER0 = 1 << ID_TC0;
541 
542  // configure TC0 channel 0 for WAVSEL=10 mode
543  // using the main clock / 2 as source
544  // and reset on C match
545  TC0->TC_CHANNEL[ 0 ].TC_CMR =
546  TC_CMR_WAVE
547  | TC_CMR_TCCLKS_TIMER_CLOCK1
548  | TC_CMR_WAVSEL_UP_RC;
549 
550  // configure pin A match clear
551  TC0->TC_CHANNEL[ 0 ].TC_CMR |= TC_CMR_ACPC_SET | TC_CMR_ACPA_CLEAR;
552 
553  // set C match/clear for 36 kHz
554  TC0->TC_CHANNEL[ 0 ].TC_RC = 41'000'000 / 36'000;
555 
556  // set 50% duty cycle
557  TC0->TC_CHANNEL[ 0 ].TC_RA = TC0->TC_CHANNEL[ 0 ].TC_RC / 2;
558 
559  write( 0 );
560 
561  // enable and start
562  TC0->TC_CHANNEL[ 0 ].TC_CCR = TC_CCR_CLKEN | TC_CCR_SWTRG;
563  }
564 
566  //
569  void write( bool b ) override {
570  if( b ){
571  // eable C match -> pin set
572  TC0->TC_CHANNEL[ 0 ].TC_CMR |= TC_CMR_ACPC_SET;
573  } else {
574  // remove pin C match action
575  TC0->TC_CHANNEL[ 0 ].TC_CMR &= ~ ( 0x3 << 18);
576  }
577  }

750 uint_fast64_t HWLIB_WEAK now_ticks(){
751  static bool init_done = false;
752  if( ! init_done ){
753 
754  // kill the watchdog
755  WDT->WDT_MR = WDT_MR_WDDIS;
756 
757  // switch to the 84 MHz crystal/PLL clock
758  sam3xa::SystemInit();
759 
760  EFC0->EEFC_FMR = EEFC_FMR_FWS(4);
761  EFC1->EEFC_FMR = EEFC_FMR_FWS(4);
762 
763  SysTick->CTRL = 0; // stop the timer
764  SysTick->LOAD = 0xFFFFFF; // use its as a 24-bit timer
765  SysTick->VAL = 0; // clear the timer
766  SysTick->CTRL = 5; // start the timer, 1:1
767 
768  init_done = true;
769  }
770 
771  static unsigned int last_low = 0;
772  static unsigned long long int high = 0;
773 
774  // the timer ticks down, but we want an up counter
775  unsigned int low = 0xFFFFFF - ( SysTick->VAL & 0xFFFFFF );
776  if( low < last_low ){
777 
778  // the timer rolled over, so increment the high part
779  high += 0x1ULL << 24;
780  }
781  last_low = low;
782 
783  // return the aggregated ticks value
784  // the counter runs at 84 MHz
785  return ( low | high );
786 
787 }

801 void uart_init(){
802  static bool init_done = false;
803  if( init_done ){
804  return;
805  }
806  init_done = true;
807 
808  // enable the clock to port A
809  PMC->PMC_PCER0 = 1 << ID_PIOA;
810 
811  // disable PIO Control on PA9 and set up for Peripheral A
812  PIOA->PIO_PDR = PIO_PA8;
813  PIOA->PIO_ABSR &= ~PIO_PA8;
814  PIOA->PIO_PDR = PIO_PA9;
815  PIOA->PIO_ABSR &= ~PIO_PA9;
816 
817  // enable the clock to the UART
818  PMC->PMC_PCER0 = ( 0x01 << ID_UART );
819 
820  // Reset and disable receiver and transmitter.
821  hw_uart->UART_CR = UART_CR_RSTRX | UART_CR_RSTTX | UART_CR_RXDIS | UART_CR_TXDIS;
822 
823  // Set the baudrate to 115200.
824  //hw_uart->UART_BRGR = 45; // MASTER_CLK_FREQ / (16 * 45) = 116666 (close enough).
825  //uart->UART_BRGR = 546; // For ~9600 baud.
826  hw_uart->UART_BRGR = 5241600 / BMPTK_BAUDRATE;
827 
828  // No parity, normal channel mode.
829  hw_uart->UART_MR = UART_MR_PAR_NO;
830 
831  // Disable all interrupts.
832  hw_uart->UART_IDR = 0xFFFFFFFF;
833 
834  // Enable the receiver and the trasmitter.
835  hw_uart->UART_CR = UART_CR_RXEN | UART_CR_TXEN;
