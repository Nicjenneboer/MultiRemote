53 void HWLIB_NORETURN kitt( port_out & port, uint_fast16_t ms ){
54  auto direct_port = direct( port );
55  for(;;){
56  for( uint_fast8_t i = 0; i < direct_port.number_of_pins(); ++i ){
57  direct_port.write( 0x01 << i );
58  wait_ms( ms );
59  }
60  for( uint_fast8_t i = direct_port.number_of_pins() - 2; i > 0; --i ){
61  direct_port.write( 0x01 << i );
62  wait_ms( ms );
63  }
64  }
65 }
