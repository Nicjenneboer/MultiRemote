134 GPIO_TypeDef & __attribute__((weak)) port_registers( uint32_t port ){
135 
136  // a bit of a cludge to put this here:
137  // enable the clock to all GPIO ports
138  RCC->APB2ENR |=
139  RCC_APB2ENR_IOPAEN | RCC_APB2ENR_IOPBEN |
140  RCC_APB2ENR_IOPCEN | RCC_APB2ENR_IOPDEN;
141 
142  switch( port ){
143  case 0 : return *GPIOA;
144  case 1 : return *GPIOB;
145  case 2 : return *GPIOC;
146  case 3 : return *GPIOD;
147  default : break;
148  }
149 
150  // doesn't return
151  HWLIB_PANIC_WITH_LOCATION;

154 class pin_base {
155 public:
156  volatile GPIO_TypeDef & port;
157  volatile uint32_t & config_word;
158  uint32_t pin;
159  uint32_t config_offset;
160  uint32_t mask;
161 
162  void config( uint32_t conf ){
163  config_word &= ~( 0xF << config_offset );
164  config_word |= conf << config_offset;
165  }
166 
167  pin_base( uint32_t port_number, uint32_t pin_number, uint32_t conf ):
168  port{ port_registers( port_number ) },
169  config_word{ ( pin_number < 8 ) ? port.CRL : port.CRH },
170  pin{ pin_number },
171  config_offset{ 4 * ( pin_number % 8 ) },
172  mask{ 0x1U << pin_number }
173  {
174  config( conf );
175  }
176 
177  bool base_read(){
178  return (( port.IDR & mask ) != 0 );
179  }
180 
181  void base_write( bool v ){
182  port.BSRR |= ( v ? mask : ( mask << 16 ));
183  }
184 
185 };

379 uint_fast64_t HWLIB_WEAK now_ticks(){
380 
381  static bool init_done = false;
382  if( ! init_done ){
383 /*
384  // switch to the 72 MHz crystal/PLL clock, from stm32x.cpp,
385  // some values taken from
386  // https://github.com/rogerclarkmelbourne/STM32duino-bootloader
387 
388  // Flash 2 wait state
389  FLASH->ACR &= (uint32_t)((uint32_t)~FLASH_ACR_LATENCY);
390  FLASH->ACR |= (uint32_t)FLASH_ACR_LATENCY_2;
391 
392  // Enable Prefetch Buffer
393  FLASH->ACR |= FLASH_ACR_PRFTBE;
394 
395  // enable HSE and wait for it
396  RCC->CR |= RCC_CR_HSEON;
397  while((RCC->CR & RCC_CR_HSERDY ) == 0 ){}
398 
399 
400  // PLL configuration: PLLCLK = HSE * 9 = 72 MHz
401  RCC->CFGR &= (uint32_t)((uint32_t)~(
402  RCC_CFGR_PLLSRC | RCC_CFGR_PLLXTPRE | RCC_CFGR_PLLMULL ));
403  RCC->CFGR |= (uint32_t)( RCC_CFGR_PLLSRC | RCC_CFGR_PLLMULL9 );
404 
405  // Enable PLL and wait for it
406  RCC->CR |= RCC_CR_PLLON;
407  while((RCC->CR & RCC_CR_PLLRDY) == 0){}
408 
409  // HCLK = SYSCLK
410  RCC->CFGR |= (uint32_t)RCC_CFGR_HPRE_DIV1;
411 
412  // PCLK2 = HCLK
413  RCC->CFGR |= (uint32_t)RCC_CFGR_PPRE2_DIV1;
414 
415  // PCLK1 = HCLK / 2
416  RCC->CFGR |= (uint32_t)RCC_CFGR_PPRE1_DIV2;
417 
418  // Select PLL as system clock source
419  RCC->CFGR &= (uint32_t)((uint32_t)~(RCC_CFGR_SW));
420  RCC->CFGR |= (uint32_t)RCC_CFGR_SW_PLL;
421 
422  // Wait till PLL is used as system clock source
423  while ((RCC->CFGR & (uint32_t)RCC_CFGR_SWS) != (uint32_t)0x08){}
424 */
425  // start the systick timer
426  SysTick->CTRL = 0; // stop the timer
427  SysTick->LOAD = 0xFFFFFF; // use its as a 24-bit timer
428  SysTick->VAL = 0; // clear the timer
429  SysTick->CTRL = 5; // start the timer, 1:1
430 
431  init_done = true;
432  }
433 
434  static unsigned int last_low = 0;
435  static unsigned long long int high = 0;
436 
437  // the timer ticks down, but we want an up counter
438  unsigned int low = 0xFFFFFF - ( SysTick->VAL & 0xFFFFFF );
439  if( low < last_low ){
440 
441  // the timer rolled over, so increment the high part
442  high += 0x1ULL << 24;
443  }
444  last_low = low;
445 
446  // return the aggregated ticks value
447  // the counter runs at 84 MHz
448  return ( low | high );
449 
450 }
