55 class line : public drawable {
56 private:
57  xy end;
58  color ink;
59  bool use_foreground;
60 
61  static void swap( int_fast16_t & a, int_fast16_t & b ){
62  auto t = a;
63  a = b;
64  b = t;
65  }
66 
67  static int_fast16_t abs( int_fast16_t x ){
68  return x >= 0 ? x : -x;
69  }
70 
71 public:
73  line( xy start, xy end, color ink )
74  : drawable{ start }, end{ end }, ink{ ink }, use_foreground( false )
75  {}
76 
78  line( xy start, xy end )
79  : drawable{ start }, end{ end }, ink{ black }, use_foreground( true )
80  {}
81 
82  void draw( window & w ) override {
83 
84  color col = use_foreground ? w.foreground : ink;
85 
86  int_fast16_t x0 = start.x;
87  int_fast16_t y0 = start.y;
88  int_fast16_t x1 = end.x;
89  int_fast16_t y1 = end.y;
90 
91  // http://en.wikipedia.org/wiki/Bresenham%27s_line_algorithm
92  // http://homepages.enterprise.net/murphy/thickline/index.html
93 
94  int_fast16_t Dx = x1 - x0;
95  int_fast16_t Dy = y1 - y0;
96 
97  int_fast16_t steep = (abs(Dy) >= abs(Dx));
98 
99  if( steep ){
100  swap( x0, y0 );
101  swap( x1, y1 );
102 
103  // recompute Dx, Dy after swap
104  Dx = x1 - x0;
105  Dy = y1 - y0;
106  }
107 
108  int_fast16_t xstep = 1;
109  if( Dx < 0 ){
110  xstep = -1;
111  Dx = -Dx;
112  }
113 
114  int_fast16_t ystep = 1;
115  if( Dy < 0 ){
116  ystep = -1;
117  Dy = -Dy;
118  }
119  int_fast16_t TwoDy = 2*Dy;
120  int_fast16_t TwoDyTwoDx = TwoDy - 2*Dx; // 2*Dy - 2*Dx
121  int_fast16_t E = TwoDy - Dx; //2*Dy - Dx
122  int_fast16_t y = y0;
123  int_fast16_t xDraw, yDraw, x;
124  for( x = x0; x != x1; x += xstep ){
125  if (steep) {
126  xDraw = y;
127  yDraw = x;
128  } else {
129  xDraw = x;
130  yDraw = y;
131  }
132 
133  w.write( xy( xDraw, yDraw ), col );
134 
135  if( E > 0 ){
136  E += TwoDyTwoDx; //E += 2*Dy - 2*Dx;
137  y = y + ystep;
138  } else {
139  E += TwoDy; //E += 2*Dy;
140  }
141  }
142  }
143 
144 }; // class line

178  void draw( window & w ) override {
179 
180  // don't draw anything when the size would be 0
181  if( radius < 1 ){
182  return;
183  }
184 
185  // http://en.wikipedia.org/wiki/Midpoint_circle_algorithm
186 
187  color col = use_foreground ? w.foreground : ink;
188 
189  int_fast16_t fx = 1 - radius;
190  int_fast16_t ddFx = 1;
191  int_fast16_t ddFy = -2 * radius;
192  int_fast16_t x = 0;
193  int_fast16_t y = radius;
194 
195  // top and bottom
196  w.write( start + xy( 0, + radius ), col );
197  w.write( start + xy( 0, - radius ), col );
198 
199  // left and right
200  w.write( start + xy( + radius, 0 ), col );
201  w.write( start + xy( - radius, 0 ), col );
202 
203  // filled circle
204  if(0) if( col != transparent ){
205 
206  // top and bottom
207  w.write( start + xy( 0, + radius ), col );
208  w.write( start + xy( 0, - radius ), col );
209 
210  // left and right
211  line(
212  start - xy( radius, 0 ),
213  start + xy( radius, 0 ),
214  col
215  ).draw( w );
216  }
217 
218  while( x < y ){
219 
220  // calculate next outer circle point
221  if( fx >= 0 ){
222  y--;
223  ddFy += 2;
224  fx += ddFy;
225  }
226  x++;
227  ddFx += 2;
228  fx += ddFx;
229 
230  w.write( start + xy( + x, + y ), col );
231  w.write( start + xy( - x, + y ), col );
232  w.write( start + xy( + x, - y ), col );
233  w.write( start + xy( - x, - y ), col );
234  w.write( start + xy( + y, + x ), col );
235  w.write( start + xy( - y, + x ), col );
236  w.write( start + xy( + y, - x ), col );
237  w.write( start + xy( - y, - x ), col );
238 
239  // filled circle
240  if(0) if( col != transparent ){
241  line(
242  start + xy( -x, y ),
243  start + xy( x, y ),
244  col ).draw( w );
245  line(
246  start + xy( -x, -y ),
247  start + xy( x, -y ),
248  col ).draw( w );
249  line(
250  start + xy( -y, x ),
251  start + xy( y, x ),
252  col ).draw( w );
253  line(
254  start + xy( -y, -x ),
255  start + xy( y, -x ),
256  col ).draw( w );
257  }
258  }
259  }
