127  // configure a pin as ADC
128  uint32_t HWLIB_WEAK configure_as_adc( uint32_t port, uint32_t pin ){
129 
130  // enable IOCON & A/D clock
131  LPC_SYSCON->SYSAHBCLKCTRL |= ( 0x01 << 16 ) | ( 0x01 << 13 );
132 
133  // enable A/D power
134  LPC_SYSCON->PDRUNCFG &= ~( 1 << 4 );
135 
136  switch( ( port << 8 ) | pin ){
137  case 0x000B: pin_ad_conf( &LPC_IOCON->R_PIO0_11, 0x02 ); return 0;
138 
139  case 0x0100: pin_ad_conf( &LPC_IOCON->R_PIO1_0, 0x02 ); return 1;
140  case 0x0101: pin_ad_conf( &LPC_IOCON->R_PIO1_1, 0x02 ); return 2;
141  case 0x0102: pin_ad_conf( &LPC_IOCON->R_PIO1_2, 0x02 ); return 3;
142  case 0x0103: pin_ad_conf( &LPC_IOCON->SWDIO_PIO1_3, 0x02 ); return 4;
143  case 0x0104: pin_ad_conf( &LPC_IOCON->PIO1_4, 0x01 ); return 5;
144 
145  default: HWLIB_PANIC_WITH_LOCATION;
146  }

321  adc_value_type read() override {
322 
323  // configure the A/D for the pin
324  LPC_ADC->CR = ( 0x01 << channel ) | ( 12 << 8 );
325 
326  // start the conversion
327  LPC_ADC->CR |= ( 0x01 << 24 );
328 
329  // wait for the conversion to complete
330  while( ( LPC_ADC->GDR & ( 1 << 31 )) == 0 );
331 
332  hwlib::wait_ms( 10 );
333 
334  // return the A/D result
335  return 3 | (( LPC_ADC->GDR >> 6 ) & 0x3FF);
336  }

364 uint_fast64_t HWLIB_WEAK now_ticks(){
365  static bool init_done = false;
366  if( ! init_done ){
367 
368  SysTick->CTRL = 0; // stop the timer
369  SysTick->LOAD = 0xFFFFFF; // use its as a 24-bit timer
370  SysTick->VAL = 0; // clear the timer
371  SysTick->CTRL = 5; // start the timer, 1:1
372 
373  init_done = true;
374  }
375 
376  static unsigned int last_low = 0;
377  static unsigned long long int high = 0;
378 
379  // the timer ticks down, but we want an up counter
380  unsigned int low = 0xFFFFFF - ( SysTick->VAL & 0xFFFFFF );
381  if( low < last_low ){
382 
383  // the timer rolled over, so increment the high part
384  high += 0x1ULL << 24;
385  }
386  last_low = low;
387 
388  // return the aggregated ticks value
389  // the counter runs at 84 MHz
390  return ( low | high );
391 
392 }
393 
394 uint64_t now_us(){
395  return now_ticks() / ticks_per_us();
396 }

402 void HWLIB_WEAK wait_us_busy( int_fast32_t n ){
403  // for the default clock frequency of 12 MHz:
404  // the loop uses 12 cycles
405  __asm volatile(
406  " mov r0, %[reg] \t\n"
407  " b 1f \t\n"
408  " .align 4 \t\n"
409  "1: sub r0, #1 \t\n"
410  " b 2f \t\n"
411  "2: b 3f \t\n"
412  "3: nop \t\n"
413  " nop \t\n"
414  " bgt 1b \t\n"
415  :: [reg] "r" (n) : "r0"
416  );
417 }
